614template <typename write_model_iterator_type>
615MONGOCXX_INLINE stdx::optional<result::bulk_write> collection::bulk_write(
616 write_model_iterator_type begin, write_model_iterator_type end,
617 const options::bulk_write& options) {
618 class bulk_write writes(options);
619
620 std::for_each(begin, end, [&](const model::write& current) { writes.append(current); });
621
622 return bulk_write(writes);
623}
