57 // findingStart mode is used to find the first operation of interest when
58 // we are scanning through a repl log. For efficiency in the common case,
59 // where the first operation of interest is closer to the tail than the head,

180 ConnectionString( ConnectionType type , const std::string& s , const std::string& setName = "" ) {
181 _type = type;
182 _setName = setName;
183 _fillServers( s );
184
185 switch ( _type ) {
186 case MASTER:
187 verify( _servers.size() == 1 );
188 break;
189 case SET:
190 verify( _setName.size() );
191 verify( _servers.size() >= 1 ); // 1 is ok since we can derive
192 break;
193 case PAIR:
194 verify( _servers.size() == 2 );
195 break;
196 default:
197 verify( _servers.size() > 0 );
198 }
199
200 _finishInit();
201 }

204 _type = INVALID;
205
206 _fillServers( s );
207 if ( _type != INVALID ) {
208 // set already
209 }
210 else if ( _servers.size() == 1 ) {
211 _type = MASTER;
212 }
213 else {
214 _type = favoredMultipleType;
215 verify( _type == SET );
216 }
217 _finishInit();
218 }
