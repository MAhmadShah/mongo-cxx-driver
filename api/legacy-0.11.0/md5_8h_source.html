78/* Define the state of the MD5 Algorithm. */
79typedef struct md5_state_s {
80 md5_word_t count[2]; /* message length in bits, lsw first */
81 md5_word_t abcd[4]; /* digest buffer */
82 md5_byte_t buf[64]; /* accumulate block */
83} md5_state_t;
