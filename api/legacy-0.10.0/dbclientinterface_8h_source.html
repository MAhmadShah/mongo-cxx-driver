58 // findingStart mode is used to find the first operation of interest when
59 // we are scanning through a repl log. For efficiency in the common case,
60 // where the first operation of interest is closer to the tail than the head,
61 // we start from the tail of the log and work backwards until we find the
62 // first operation of interest. Then we scan forward from that first operation,
63 // actually returning results to the client. During the findingStart phase,
64 // we release the db mutex occasionally to avoid blocking the db process for

181 ConnectionString( ConnectionType type , const std::string& s , const std::string& setName = "" ) {
182 _type = type;
183 _setName = setName;
184 _fillServers( s );
185
186 switch ( _type ) {
187 case MASTER:
188 verify( _servers.size() == 1 );
189 break;
190 case SET:
191 verify( _setName.size() );
192 verify( _servers.size() >= 1 ); // 1 is ok since we can derive
193 break;
194 case PAIR:
195 verify( _servers.size() == 2 );
196 break;
197 default:
198 verify( _servers.size() > 0 );
199 }
200
201 _finishInit();
202 }

205 _type = INVALID;
206
207 _fillServers( s );
208 if ( _type != INVALID ) {
209 // set already

211 else if ( _servers.size() == 1 ) {
212 _type = MASTER;
213 }
214 else {
215 _type = favoredMultipleType;
216 verify( _type == SET );
217 }
218 _finishInit();
219 }
